40  /**
41   * Enforcer rule to ban vulnerable dependencies.
42   *
43   * @since ???
44   */
45  public class BanVulnerableDependencies
46      extends EnforcerRuleSupport
47  {
48    /**
49     * <a href="https://ossindex.sonatype.org/">Sonatype OSS Index</a> client configuration.
50     */
51    private OssindexClientConfiguration clientConfiguration = new OssindexClientConfiguration();
52  
53    /**
54     * Limit scope of dependency resolution.
55     */
56    private String scope;
57  
58    /**
59     * Include transitive dependencies.
60     */
61    private boolean transitive = true;
62  
63    /**
64     * Set of coordinates to exclude from vulnerability matching.
65     */
66    private Set<MavenCoordinates> excludeCoordinates = new HashSet<>();
67  
68    /**
69     * CVSS-score threshold.  Vulnerabilities with lower scores will be excluded.
70     */
71    private float cvssScoreThreshold = 0;
72  
73    /**
74     * Set of <a href="https://ossindex.sonatype.org/">Sonatype OSS Index</a>
75     * vulnerability identifiers to exclude from matching.
76     */
77    private Set<String> excludeVulnerabilityIds = new HashSet<>();
78  
79    public OssindexClientConfiguration getClientConfiguration() {
80      return clientConfiguration;
81    }
82  
83    public void setClientConfiguration(final OssindexClientConfiguration clientConfiguration) {
84      this.clientConfiguration = clientConfiguration;
85    }
86  
87    public String getScope() {
88      return scope;
89    }
90  
91    public void setScope(final String scope) {
92      this.scope = scope;
93    }
94  
95    public boolean isTransitive() {
96      return transitive;
97    }
98  
99    public void setTransitive(final boolean transitive) {
100     this.transitive = transitive;
101   }

115   }

119   }

122     return excludeVulnerabilityIds;
123   }
124

126     this.excludeVulnerabilityIds = excludeVulnerabilityIds;

129   @Override
130   public void execute(@Nonnull final EnforcerRuleHelper helper) throws EnforcerRuleException {
131     new Task(helper).run();
132   }
133 
134   /**
135    * Encapsulates state for rule evaluation.
136    */
137   private class Task
138   {
139     private final Log log;
140 
141     private final MavenSession session;
142 
143     private final MavenProject project;
144 
145     private final DependencyGraphBuilder graphBuilder;
146 
147     private final ComponentReportAssistant reportAssistant;
148 
149     public Task(final EnforcerRuleHelper helper) {
150       this.log = helper.getLog();
151       this.session = lookup(helper, "${session}", MavenSession.class);
152       this.project = lookup(helper, "${project}", MavenProject.class);
153       this.graphBuilder = lookup(helper, DependencyGraphBuilder.class);
154       this.reportAssistant = lookup(helper, ComponentReportAssistant.class);
155     }
156 
157     // FIXME: adjust all logging to include rule-simple-name; what does this show up as by default?
158 
159     public void run() throws EnforcerRuleException {
160       // skip if maven is in offline mode
161       if (session.isOffline()) {
162         log.warn("Skipping " + BanVulnerableDependencies.class.getSimpleName() + "; offline");
163         return;
164       }
165 
166       // skip if packaging is pom
167       if ("pom".equals(project.getPackaging())) {
168         log.debug("Skipping; POM module");
169         return;
170       }
171 
172       // determine dependencies
173       Set<Artifact> dependencies;
174       try {
175         dependencies = resolveDependencies();
176       }
177       catch (DependencyGraphBuilderException e) {
178         throw new RuntimeException("Failed to resolve dependencies", e);
179       }
180 
181       // skip if project has no dependencies
182       if (dependencies.isEmpty()) {
183         log.debug("Skipping; zero dependencies");
184         return;
185       }
186 
187       ComponentReportRequest reportRequest = new ComponentReportRequest();
188       reportRequest.setClientConfiguration(clientConfiguration);
189       reportRequest.setExcludeCoordinates(excludeCoordinates);
190       reportRequest.setExcludeVulnerabilityIds(excludeVulnerabilityIds);
191       reportRequest.setCvssScoreThreshold(cvssScoreThreshold);
192       reportRequest.setComponents(dependencies);
193 
194       ComponentReportResult reportResult = reportAssistant.request(reportRequest);
195 
196       if (reportResult.hasVulnerable()) {
197         throw new EnforcerRuleException(reportResult.explain());
198       }
199     }
200 
201     /**
202      * Resolve dependencies to inspect for vulnerabilities.
203      */
204     private Set<Artifact> resolveDependencies() throws DependencyGraphBuilderException {
205       Set<Artifact> result = new HashSet<>();
206 
207       ArtifactFilter artifactFilter = null;
208       if (scope != null) {
209         List<String> scopes = Splitter.on(',').trimResults().omitEmptyStrings().splitToList(scope);
210         artifactFilter = new CumulativeScopeArtifactFilter(scopes);
211       }
212 
213       DependencyNode node = graphBuilder.buildDependencyGraph(project, artifactFilter);
214       collectArtifacts(result, node);
215 
216       return result;
217     }
218 
219     /**
220      * Collect artifacts from dependency.
221      *
222      * Optionally including transitive dependencies if {@link #transitive} is {@code true}.
223      */
224     @SuppressWarnings("Duplicates")
225     private void collectArtifacts(final Set<Artifact> artifacts, final DependencyNode node) {
226       if (node.getChildren() != null) {
227         for (DependencyNode child : node.getChildren()) {
228           artifacts.add(child.getArtifact());
229 
230           if (transitive) {
231             collectArtifacts(artifacts, child);
232           }
233         }
234       }
235     }
236   }
